```ruby
def setup
  @rag_config = {
    vector_db: {
      provider: 'chromadb',
      mode: 'in_memory'  # Isolated from production
    },
    embedding_service: {
      provider: 'ollama',
      model: 'nomic-embed-text'  # Small, fast model for testing
    },
    collection_name: 'test_rag_collection'  # Separate from production

end
```
